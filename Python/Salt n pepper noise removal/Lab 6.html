import cv2
import numpy as np
import matplotlib.pyplot as plt

hubble = cv2.imread('3.png', 0)

kernel = np.ones((15,15), np.float32)/225
dst = cv2.filter2D(hubble, -1, kernel)
 
res,thresh = cv2.threshold(dst, 110, 255, cv2.THRESH_BINARY)

result = np.hstack((hubble, dst, thresh))
cv2.imshow('result', result)

cv2.waitKey(0)

plt.imshow(cv2.cvtColor(hubble, cv2.COLOR_BGR2RGB))
plt.show()
